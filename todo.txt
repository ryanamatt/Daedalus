to_hessenberg(A):
    n = A.shape[0]
    H = copy(A)

    for k in 0..n-2:
        // Same Householder logic as above, but applied to H[k+1..n, k]
        // (one row lower than in full QR)
        x = H[k+1..n, k]

        alpha = -sign(x[0]) * norm(x)
        v = copy(x)
        v[0] -= alpha
        if norm(v) < 1e-12: continue
        v /= norm(v)

        // Apply from left  → zeros out below sub-diagonal in column k
        H[k+1..n, k..n] -= 2 * outer(v, vᵀ * H[k+1..n, k..n])
        // Apply from right → preserves eigenvalues (similarity transform)
        H[0..n, k+1..n] -= 2 * outer(H[0..n, k+1..n] * v, vᵀ)

    return H
    // H is similar to A (same eigenvalues), but upper-Hessenberg

householder_qr(A):
    m, n = A.shape
    Q = identity(m)
    R = copy(A)

    for k in 0..min(m-1, n):
        // Extract the column vector below (and including) the diagonal
        x = R[k..m, k]          // a sub-column vector of length (m - k)

        // Build the reflection vector v
        // The sign trick avoids cancellation when x[0] is already large
        alpha = -sign(x[0]) * norm(x)
        v = copy(x)
        v[0] = v[0] - alpha     // v = x - alpha*e1

        if norm(v) < 1e-12: continue   // already zero, skip

        v = v / norm(v)         // normalize

        // Apply the Householder reflector H = I - 2*v*vᵀ to R and Q
        // We never form H explicitly — instead we apply it as rank-1 updates
        // R[k..m, k..n] = R[k..m, k..n] - 2 * v * (vᵀ * R[k..m, k..n])
        // Q[:, k..m]    = Q[:, k..m]    - 2 * (Q[:, k..m] * v) * vᵀ

        sub_R = R[k..m, k..n]
        R[k..m, k..n] = sub_R - 2 * outer(v, v @ sub_R)

        sub_Q = Q[:, k..m]        // columns k to m of Q
        Q[:, k..m] = sub_Q - 2 * outer(sub_Q @ v, v)

    return Q, R
    // Invariant: A == Q * R,  Q is orthogonal,  R is upper-triangular

qr_iteration(A, tol=1e-10, max_iter=1000):
    n = A.shape[0]
    H = copy(A)           // work on a copy; reduce to Hessenberg first (see note)
    eigenvalues = zeros(n)

    // Deflation loop: shrink the active submatrix as eigenvalues converge
    active_size = n
    
    for iter in 0..max_iter:
        if active_size == 1:
            eigenvalues[0] = H[0, 0]
            break

        // --- Wilkinson shift ---
        // Look at the bottom-right 2x2 corner of the active submatrix
        a = H[active_size-2, active_size-2]
        b = H[active_size-2, active_size-1]
        c = H[active_size-1, active_size-1]
        
        // Eigenvalue of the 2x2 that is closer to H[active-1, active-1]
        delta = (a - c) / 2.0
        sign_delta = +1 if delta >= 0 else -1
        shift = c - sign_delta * b² / (abs(delta) + sqrt(delta² + b²))

        // --- Shifted QR step ---
        Q, R = householder_qr(H[0..active_size, 0..active_size] - shift * I)
        H[0..active_size, 0..active_size] = R * Q + shift * I

        // --- Check for convergence (deflation) ---
        // If the sub-diagonal entry at the bottom is near zero,
        // the last diagonal entry has converged to an eigenvalue
        if abs(H[active_size-1, active_size-2]) < tol * (abs(H[active_size-2, active_size-2]) + abs(H[active_size-1, active_size-1])):
            eigenvalues[active_size - 1] = H[active_size-1, active_size-1]
            active_size -= 1    // peel off the converged eigenvalue

    return sort(eigenvalues, descending=True)
